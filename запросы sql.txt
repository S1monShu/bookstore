books_author.id_book = books.id) AS 'Авторы',

(SELECT GROUP_CONCAT(genres.genre SEPARATOR ', ') FROM genres, books_genres WHERE
genres.id_genres = books_genres.id_genre AND
books_genres.id_book = books.id) AS 'Жанры'

FROM books
